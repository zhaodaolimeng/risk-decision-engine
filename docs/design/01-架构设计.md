### 2.2 核心模块

#### 2.2.1 规则引擎

**职责**：
- 规则解析
- 规则执行
- 规则校验

**核心接口**：

```go
// RuleEngine 规则引擎接口
type RuleEngine interface {
    // Execute 执行规则
    Execute(ctx context.Context, rule *Rule, fact *Fact) (*RuleResult, error)

    // ExecuteBatch 批量执行规则
    ExecuteBatch(ctx context.Context, rules []*Rule, fact *Fact) ([]*RuleResult, error)
}

// Rule 规则定义
type Rule struct {
    ID          string
    Name        string
    Description string
    Type        RuleType  // BOOLEAN, NUMERIC
    Condition   Expression
    Actions     []Action
    Priority    int
    Version     string
}

// Fact 事实数据
type Fact struct {
    Data map[string]interface{}
}

// RuleResult 规则执行结果
type RuleResult struct {
    RuleID      string
    RuleName    string
    Pass        bool
    Hit         bool
    Output      map[string]interface{}
    Reason      string
    ExecuteTime time.Duration
}
```

**规则示例（JSON）**：

```json
{
  "id": "R001",
  "name": "年龄准入规则",
  "description": "申请人年龄必须在18-60岁之间",

  "condition": {
    "operator": "AND",
    "expressions": [
      {
        "field": "age",

        "value": 18
      },
      {
        "field": "age",

    "true": {

  "version": "1.0"
```

#### 2.2.2 决策流引擎

**职责**：
- 决策流编排
- 节点执行
- 流程控制
- 上下文数据传递

**核心接口**：

```go
// FlowEngine 决策流引擎接口
type FlowEngine interface {
    // Execute 执行决策流
    Execute(ctx context.Context, flow *Flow, fact *Fact) (*FlowResult, error)
}

// Flow 决策流定义
type Flow struct {
    ID          string
    Name        string
    Description string
    Nodes       []Node
    Edges       []Edge
    StartNodeID string
    Version     string
}

// Node 节点
type Node interface {
    GetID() string
    GetType() NodeType
    Execute(ctx context.Context, fact *Fact) (*NodeResult, error)
}

// NodeType 节点类型
const (
    NodeTypeStart    NodeType = "START"
    NodeTypeEnd      NodeType = "END"
    NodeTypeRuleSet  NodeType = "RULE_SET"
    NodeTypeModel    NodeType = "MODEL"
    NodeTypeDecision NodeType = "DECISION"
    NodeTypeSplit    NodeType = "SPLIT"
)

// Edge 连线
type Edge struct {
    FromNodeID string
    ToNodeID   string
    Condition  Expression
}

// FlowResult 决策流结果
type FlowResult struct {
    FlowID      string
    FlowName    string
    Decision    string
    Reason      string
    NodeResults []*NodeResult
    ExecuteTime time.Duration
}
```

```go
type DataSource interface {
```

#### 2.2.4 模型服务

**职责**：
- 模型注册和管理
- 模型调用（HTTP）
- 上下文数据传递
- 模型结果解析和映射

**核心接口**：

```go
// ModelEngine 模型引擎接口
type ModelEngine interface {
    // Execute 执行模型
    Execute(ctx context.Context, model *Model, input *ModelInput) (*ModelResult, error)
}

// Model 模型定义
type Model struct {
    ID             string
    Name           string
    Description    string
    Type           ModelType  // HTTP
    Config         ModelConfig
    InputSchema    map[string]interface{}
    OutputSchema   map[string]interface{}
    Version        string
    Status         ModelStatus
}

// ModelType 模型类型
const (
    ModelTypeHTTP ModelType = "HTTP"
)

// ModelConfig 模型配置
type ModelConfig struct {
    URL            string                 // 模型服务URL
    Method         string                 // HTTP方法：POST/GET
    Headers        map[string]string      // 请求头
    Timeout        time.Duration          // 超时时间
    InputMapping   map[string]string      // 输入字段映射（模板）
    OutputMapping  map[string]string      // 输出字段映射
    RetryConfig    *RetryConfig           // 重试配置
}

// RetryConfig 重试配置
type RetryConfig struct {
    MaxAttempts  int
    RetryDelay   time.Duration
}

// ModelInput 模型输入
type ModelInput struct {
    ContractID    string                 // 合同ID
    ContextData   map[string]interface{} // 上下文数据（预加载的数据）
    RequestData   map[string]interface{} // 请求数据
}

// ModelResult 模型结果
type ModelResult struct {
    ModelID       string
    ModelName     string
    Success       bool
    Output        map[string]interface{} // 解析后的输出
    RawOutput     interface{}            // 原始输出
    ErrorMessage  string
    ExecuteTime   time.Duration
}

// ModelService 模型服务
type ModelService interface {
    Register(model *Model) error
    GetModel(modelID string) (*Model, error)
    ListModels() ([]*Model, error)
    InvokeModel(ctx context.Context, modelID string, input *ModelInput) (*ModelResult, error)
}
```
